Weight Object
===============================================
A Weight object describes the weight of an item.



Properties
-----------------------------------------------

<Reference>

  <Field name="value" type="number" required={true}>
    <Description>
      The weight value for this package. This value may contain decimals.
    </Description>
  </Field>

  <Field name="unit" type="string" required={true}>
    <Description>
      The unit of measure for this weight. Valid values include the following:
      * `g` - grams
      * `oz` - ounces
      * `kg` - kilograms
      * `lb` - pounds
    </Description>
  </Field>

  <Field name="ounces" type="number" required={true}>
    <Description>
      The weight in ounces.
    </Description>
  </Field>

  <Field name="grams" type="number" required={true}>
    <Description>
      The weight in grams.
    </Description>
  </Field>

  <Field name="toOunces()" required={true}>
    <Type>
      method
    </Type>
    <Description>
      A method that returns the weight in ounces.
    </Description>
  </Field>

  <Field name="toGrams()" required={true}>
    <Type>
      method
    </Type>
    <Description>
      A method that returns the weight in grams.
    </Description>
  </Field>

</Reference>
