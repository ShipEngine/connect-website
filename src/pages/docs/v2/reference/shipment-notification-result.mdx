ShipmentNotificationResult
===============================================
The result of an individual shipment notification request

<Reference>
  <Field name="confirmation_code" type="string" nullable={true}>
    <Description>
      An optional confirmation code associated with this notification
    </Description>
  </Field>
  <Field name="failure_reason" type="string" nullable={true}>
    <Description>
      If succeeded was false use this field to indicate why
    </Description>
  </Field>
  <Field name="notification_id" type="string" nullable={false}>
    <Description>
      The notification id from the request
    </Description>
  </Field>
  <Field name="status" nullable={true}>
    <Type>
      [NotificationStatus](./notification-status.mdx)
    </Type>
    <Description>
      Indicates the status for the third party receiving this notification
    </Description>
  </Field>
  <Field name="submission_id" type="string" nullable={true}>
    <Description>
      A unique identifier that is used to query the third party about the status of this notification in the notification_status endpoint
    </Description>
  </Field>
  <Field name="succeeded" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this request was successful.
    </Description>
  </Field>
</Reference>
