CancelSalesOrderItemsResult
===============================================
Cancel result item

<Reference>
  <Field name="confimation_code" type="string" nullable={true}>
    <Description>
      An identifier provided by the order source to cancel the items on the fulfillment
    </Description>
  </Field>
  <Field name="failure_reason" type="string" nullable={true}>
    <Description>
      Details given by the order source if the cancel action fails
    </Description>
  </Field>
  <Field name="notification_id" type="string" nullable={false}>
    <Description>
      Identifier for the requested cancel notification
    </Description>
  </Field>
  <Field name="status" nullable={false}>
    <Type>
      string
    </Type>
    <Description>
      Result status from the cancel action performed on the item. Only supported values are 'success' or 'failure'.
    </Description>
  </Field>
</Reference>
