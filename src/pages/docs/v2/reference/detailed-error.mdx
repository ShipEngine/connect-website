DetailedError
===============================================
This represents information that can be gathered about an error

<Reference>
  <Field name="external_error_code" type="string" nullable={true}>
    <Description>
      This is the custom error code used by external systems @example "AMAZON_ERR_124", "shopify_bad_request_435323"
    </Description>
  </Field>
  <Field name="external_http_status_code" type="number" nullable={true}>
    <Description>
      The status code associated with this error
    </Description>
  </Field>
  <Field name="message" type="string" nullable={true}>
    <Description>
      An error message that will be bubbled up to the users
    </Description>
  </Field>
  <Field name="raw_external_context" type="string" nullable={true}>
    <Description>
      The full response or error returned (serialized)
    </Description>
  </Field>
  <Field name="standardized_error_code" nullable={false}>
    <Type>
      [ShipEngineErrorCode](./ship-engine-error-code.mdx)
    </Type>
    <Description>
      The standarized code associated with this error type
    </Description>
  </Field>
</Reference>
