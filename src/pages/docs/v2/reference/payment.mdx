Payment
===============================================
This represents a payment made for an order

<Reference>
  <Field name="adjustments" nullable={true}>
    <Type>
      [Charge](./charge.mdx)[]
    </Type>
    <Description>
      A list of adjustments applied that influence the order total. For example, promotions/discounts/coupons. The amount should always be a quantity of currency, not a percentage.
    </Description>
  </Field>
  <Field name="amount_paid" type="number" nullable={true}>
    <Description>
      The amount of the currency
    </Description>
  </Field>
  <Field name="coupon_code" nullable={true}>
    <Type>
      string[]
    </Type>
    <Description>
      Coupon codes applied to obtain a discount. The discount amounts should be included as one of the adjustments.
    </Description>
  </Field>
  <Field name="label_voucher" nullable={true}>
    <Type>
      [LabelVoucher](./label-voucher.mdx)
    </Type>
    <Description>
      Represents information needed to leverage a third party Carrier API implementation
    </Description>
  </Field>
  <Field name="payment_id" type="string" nullable={true}>
    <Description>
      An ID for this payment in the vendor API
    </Description>
  </Field>
  <Field name="payment_method" type="string" nullable={true}>
    <Description>
      The payment method
    </Description>
  </Field>
  <Field name="payment_status" nullable={true}>
    <Type>
      [PaymentStatus](./payment-status.mdx)
    </Type>
    <Description>
      The status of whether or not an order has been paid for
    </Description>
  </Field>
  <Field name="shipping_charges" nullable={true}>
    <Type>
      [Charge](./charge.mdx)[]
    </Type>
    <Description>
      A list of shipping charges.
    </Description>
  </Field>
  <Field name="taxes" nullable={true}>
    <Type>
      [Charge](./charge.mdx)[]
    </Type>
    <Description>
      A list of tax charges. The description can convey the jurisdiction
    </Description>
  </Field>
</Reference>
