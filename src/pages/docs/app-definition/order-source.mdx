# Order Source App Definition

The code for your app definition can be seen [here on github](https://github.com/ShipEngine/connect/blob/master/packages/order-source-api/src/app/ordersource-app-definition.ts).

## Metadata

<Reference>
    <Field name="Id" type="GUID" nullable={false} required={true}>
    <Description>
        The id for this integration **(this GUID should not be changed after
        publishing)**
    </Description>
    </Field>

    <Field name="Name" type="string" nullable={false} required={true}>
    <Description>
      The name of this integration, used for internal purposes, can be the same as the branded OrderSource in OrderSources[]
    </Description>
    </Field>

    <Field name="AuthProcess" type="object" nullable={false} required={true}>
    <Description>
      The [specification](#auth-specification) for authorizing with this order source
    </Description>
    </Field>

    <Field name="OrderSources[]" type="object[]" nullable={false} required={true}>
    <Description>
      A list of branded [order sources](#order-source)
    </Description>
    </Field>

</Reference>

## Order Source

<Reference>
        <Field name="Id" type="GUID" nullable={false} required={true}>
    <Description>
      This is a unique GUID identifier for this order source **(this GUID should not be changed after
        publishing)**
    </Description>
    </Field>

    <Field name="Name" type="string" nullable={false} required={true}>
    <Description>
      The branded name for this order source "Shopify", "eBay", "Amazon" **(This is what our customers see)**
    </Description>
    </Field>

    <Field name="SendEmail" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not our system is allowed to send emails to customers
    </Description>
    </Field>

    <Field name="HasCustomMappings" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not this marketplace allows custom mappings
    </Description>
    </Field>

    <Field name="CanRefresh" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not orders can be refreshed
    </Description>
    </Field>

    <Field name="CanLeaveFeedback" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not the order source allows sellers to leave feedback on customers
    </Description>
    </Field>

    <Field name="CanConfirmShipments" type="boolean" nullable={false} required={true}>
    <Description>
        Indicates whether or not the order source can confirm an order has been shipped
    </Description>
    </Field>

    <Field name="IsRefreshDisabled" type="boolean" nullable={true} required={false}>
    <Description>
      Indicates that refreshing is disabled within our platform
    </Description>
    </Field>

    <Field name="HasCustomStatuses" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether this order source allows for custom statuses on a seller by seller basis
    </Description>
    </Field>

    <Field name="CanConfigureTimeZone" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not a user is allowed to configure their own timezone
    </Description>
    </Field>

    <Field name="HasInventoryLevels" type="boolean" nullable={false} required={true}>
    <Description>
      Indicates whether or not the order source has inventory levels
    </Description>
    </Field>

    <Field name="AccountConnection" type="object" nullable={false} required={true}>
    <Description>
      Specifies account connection information
    </Description>
    </Field>
    <Field name="AccountConnection.Name" type="string" nullable={false} required={true}>
    <Description>
      The name of your connection form
    </Description>
    </Field>
    <Field name="AccountConnection.ConnectionFormSchema" nullable={false} required={true}>
    <Type>
      [Form](./../reference/forms.mdx)
    </Type>
    <Description>
      A form that allows the user to specify the credentials needed to generate the OAuth token.
    </Description>
    </Field>

    <Field name="Images" type="object" nullable={false} required={true}>
    <Description>
      Images that will be used for this branded order source
    </Description>
    </Field>

    <Field name="Images.Logo" type="string" nullable={false} required={true}>
    <Description>
      The full path to the logo used in modals and other areas of our platform for this order source. Use join(__dirname, '../assets/logo.svg') "/dev/integration/assets/ordersource1/logo.svg"
    </Description>
    </Field>

    <Field name="Images.Icon" type="string" nullable={false} required={true}>
    <Description>
      The full path to the icon used for this order source. Use join(__dirname, '../assets/logo.svg')  "/dev/integration/assets/ordersource1/icon.svg"
    </Description>
    </Field>

</Reference>
