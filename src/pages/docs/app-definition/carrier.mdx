# Carrier App Definition

## Metadata

<Reference>
    <Field name="Id" type="GUID" nullable={false} required={true}>
    <Description>
        The id for this integration **(this GUID should not be changed after
        publishing)**
    </Description>
    </Field>

    <Field name="Name" type="string" nullable={false} required={true}>
    <Description>
      The name of this integration, used for internal purposes, can be the same as the branded Carrier in Carriers[]
    </Description>
    </Field>

    <Field name="AuthProcess" type="object" nullable={true} required={false}>
    <Description>
      The [specification](#auth-specification) for authorizing with this carrier
    </Description>
    </Field>

    <Field name="Carriers[]" type="object[]" nullable={false} required={true}>
    <Description>
      A list of branded [carriers](#carrier)
    </Description>
    </Field>
</Reference>

## Carrier

<Reference>
    <Field name="Id" type="GUID" nullable={false} required={true}>
    <Description>
        The id for this branded carrier **(this GUID should not be changed after
        publishing)**
    </Description>
    </Field>

    <Field name="Name" type="string" nullable={false} required={true}>
    <Description>
      The name of this branded carrier, this is what is seen by the customer
    </Description>
    </Field>

    <Field name="ApiCode" type="string" nullable={true} required={false}>
    <Description>
      The identifier for this carrier to be used in API calls. Must be snake cased and unique.
    </Description>
    </Field>

    <Field name="Description" type="string" nullable={true} required={false}>
    <Description>
      This is a human readable description of the branded carrier
    </Description>
    </Field>

    <Field name="CarrierUrl" type="string" nullable={true} required={false}>
    <Description>
      This is the url to the carriers website
    </Description>
    </Field>

    <Field name="TrackingUrl" type="string" nullable={true} required={false}>
    <Description>
      This is the url template for static tracking pages ex: `http://www.shipper.com/tracking/{0}`
    </Description>
    </Field>

    <Field name="CarrierAttributes[]" type="string[]" nullable={true} required={false}>
    <Description>
      Attributes describing the carrier ex: `ManifestDigital`, `ManifestPhysical`, `Consolidator`, `Regional`
    </Description>
    </Field>

    <Field name="DefaultConfirmationTypes" nullable={true} required={false}>
    <Type>
      [ConfirmationDictionary](./../reference/confirmation-dictionary.mdx)
    </Type>
    <Description>
      This dictionary allows you to specify which supported delivery confirmation types the carrier offers. Setting one to undefined will use the default name in shipstation. 
    </Description>
    </Field>

    <Field name="LabelFormats" nullable={true} required={false}>
      <Type>
      [LabelFormatsEnum[]](./../reference/label-formats-enum.mdx)
    </Type>
    <Description>
      This provides a list of label formats supported by the carrier
    </Description>
    </Field>

    <Field name="DefaultLabelSizes" nullable={true} required={false}>
      <Type>
      [LabelSizesEnum[]](./../reference/label-sizes-enum.mdx)
    </Type>
    <Description>
      This provides a list of supported label sizes by the carrier
    </Description>
    </Field>

    <Field name="DefaultSupportedCountries[]" type="object[]" nullable={true} required={false}>
    <Description>
      A list of supported countries, used to determine which services are visible based on shipstation sellers home country
    </Description>
    </Field>
    <Field name="DefaultSupportedCountries[].FromCountry" type="string" nullable={false} required={true}>
    <Description>
      The 2 character ISO country code ex: `US`, `MX`, `FR`
    </Description>
    </Field>
    <Field name="DefaultSupportedCountries[].AlternateServiceName" type="string" nullable={true} required={false}>
    <Description>
      An optional name for this service based on the countries branding
    </Description>
    </Field>

    <Field name="ShippingOptions" type="object" nullable={true} required={false}>
    <Description>
      This object defines which shipping options should be made available and what those shipping options should be called in the ui
    </Description>
    </Field>

    <Field name="ShippingServices[]" type="object[]" nullable={false} required={true}>
    <Description>
      A list of [shipping services](#shipping-service) associated with this branded carrier
    </Description>
    </Field>

    <Field name="PackageTypes[]" type="object[]" nullable={true} required={false}>
    <Description>
      A list of [package types](#package-type) associated with this branded carrier
    </Description>
    </Field>

    <Field name="AccountModals" type="object" nullable={false} required={true}>
    <Description>
      This defines the connection modal and the settings modals in the ui, along with branded images.
    </Description>
    </Field>

    <Field name="AccountModals.RegistrationFormSchema" nullable={false} required={true}>
    <Type>
      [Form](./../reference/forms.mdx)
    </Type>
    <Description>
      A form that allows the user to connect to the service. This form will usually prompt for an account number and login credentials.
    </Description>
    </Field>

    <Field name="AccountModals.SettingsFormSchema" nullable={false} required={true}>
    <Type>
      [Form](./../reference/forms.mdx)
    </Type>
    <Description>
      	A form that allows the user update their connection settings, such as when a password is changed.
    </Description>
    </Field>
    
    <Field name="Images" type="object" nullable={false} required={true}>
    <Description>
      This defines a list of images associated with this branded carrier
    </Description>
    </Field>
    <Field name="Images.Logo" type="string" nullable={false} required={true}>
    <Description>
      This is the full file path to the branded logo
    </Description>
    </Field>
    <Field name="Images.Icon" type="string" nullable={false} required={true}>
    <Description>
      This is the full file path to the branded icon
    </Description>
    </Field>
    
</Reference>

## Package Type

<Reference>
    <Field name="Id" type="GUID" nullable={false} required={true}>
    <Description>
      The unique identifier for this package type. **It should not change after being published**
    </Description>
    </Field>

    <Field name="Name" type="string" nullable={false} required={true}>
    <Description>
      The name of this package type ex: 'Eco friendly box'
    </Description>
    </Field>

    <Field name="ApiCode" type="string" nullable={true} required={false}>
    <Description>
      The identifier for this package type to be used in API calls. Must be snake cased and unique.
    </Description>
    </Field>

    <Field name="CarrierPackageTypeCode" type="string" nullable={false} required={true}>
    <Description>
      This is the code that will be sent to requests, this should be the same code the carrier's api would expect
    </Description>
    </Field>

    <Field name="Description" type="string" nullable={true} required={false}>
    <Description>
      This is a human readable description about what the packaging is
    </Description>
    </Field>

    <Field name="Abbreviation" type="string" nullable={true} required={false}>
    <Description>
      This is an abbreviation of the name if necessary
    </Description>
    </Field>

    <Field name="PackageAttributes" type="string[]" nullable={false} required={true}>
    <Description>
      Package attributes include `International`, `Domestic`, `Consolidator`
    </Description>
    </Field>

    <Field name="RequiredToShip" type="string[]" nullable={true} required={false}>
    <Description>
      What information is required to ship this package type `Weight` and or `Dimensions`
    </Description>
    </Field>

</Reference>
