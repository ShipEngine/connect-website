Sales Order Item Object
===============================================
Sales order item objects are used throughout [ShipEngine Connect](./../../index.mdx) to describe single items in a sales order.


Properties
-----------------------------------------------
This table lists the properties of a sales order item object and identifies those properties that are required. This object
is used for a property of the object returned from the [`getSalesOrdersByDate` method](./methods/get-sales-orders-by-date.mdx) method and is
not passed as an argument to any method.

<Reference>

  <Field name="id" type="string" required={true}>
    <Description>
      The marketplace's unique ID for the order item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="name" type="string" required={true}>
    <Description>
      The user-friendly name of the item. This is often the same as the product name. This string must not contain newline characters.
    </Description>
  </Field>

<Field name="product" type="object" required={true}>
    <Description>
      The product associated with this item.
    </Description>
  </Field>

  <Field name="product.id" type="string" required={true}>
    <Description>
      The marketplace's unique ID for the product. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.identifiers" required={false}>
    <Description>
      Your own identifiers for this product.
    </Description>
    <Type>
      [identifiers object](./identifiers-object.mdx)
    </Type>
  </Field>

  <Field name="product.details" required={false}>
    <Description>
      A list of details associated with this product.
    </Description>
    <Type>
      [identifiers object](./identifiers-object.mdx)
    </Type>
  </Field>

  <Field name="product.sku" type="string" required={false}>
    <Description>
      The [Stock Keeping Unit](https://en.wikipedia.org/wiki/Stock_keeping_unit). This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.upc" type="string" required={false}>
    <Description>
      The [Universal Product Code](https://en.wikipedia.org/wiki/Universal_Product_Code) for this item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.isbn" type="string" required={false}>
    <Description>
      The [International Standard Book Number](https://www.isbn-international.org/) for this item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.asin" type="string" required={false}>
    <Description>
      The [Amazon Standard Identification Number](https://www.amazon.com/gp/seller/asin-upc-isbn-info.html) for this item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.fulfillmentSku" type="string" required={false}>
    <Description>
      The [Stock Keeping Unit](https://en.wikipedia.org/wiki/Stock_keeping_unit) related to the fulfillment of this item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.inventoryID" type="string" required={false}>
    <Description>
      The inventory ID for this item. This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="product.dimensions" type="object" required={false}>
     <Description>
       The dimensions for the product.
     </Description>
  </Field>

  <Field name="product.dimensions.length" type="number" required={false}>
    <Description>
      The length of the product. This value may contain decimals.
    </Description>
  </Field>

  <Field name="product.dimensions.width" type="number" required={false}>
    <Description>
      The width of the product. This value may contain decimals.
    </Description>
  </Field>

  <Field name="product.dimensions.height" type="number" required={false}>
    <Description>
      The height of the product. This value may contain decimals.
    </Description>
  </Field>

  <Field name="product.dimensions.unit" type="string" required={true}>
    <Description>
      The unit of measurement for the dimensions. Valid values include the following:
      * `in` - inches
      * `cm` - centimeters
    </Description>
  </Field>

  <Field name="quantity" type="object" required={true}>
    <Description>
      The quantity of this item in the sales order.
    </Description>
  </Field>

  <Field name="quantity.value" type="number" required={true}>
    <Description>
      The quantity of items in this sales order item. The minimum value for this property is `1`.
    </Description>
  </Field>

  <Field name="unitPrice" type="object" required={true}>
    <Description>
      The sale price of each item. This should NOT include additional charges or adjustments, such as taxes or discounts.
    </Description>
  </Field>

  <Field name="unitPrice.value" type="number" required={true}>
    <Description>
      The value for this amount.
    </Description>
  </Field>

  <Field name="unitPrice.currency" type="string" required={true}>
    <Description>
      The currency for this amount.
    </Description>
  </Field>

  <Field name="unitWeight" type="object" required={false}>
    <Description>
      The weight of each item.
    </Description>
  </Field>

  <Field name="unitWeight.value" type="number" required={true}>
    <Description>
      The value for the weight. The minimum value for this property is `1`.
    </Description>
  </Field>

  <Field name="unitWeight.unit" type="string" required={true}>
    <Description>
      The unit of measure for this weight. Valid values include the following:
        * `g` for grams
        * `oz` for ounces
        * `kg` for kilograms
        * `lb` for pounds
    </Description>
  </Field>

  <Field name="sku" type="string" required={false}>
    <Description>
      The [Stock Keeping Unit](https://en.wikipedia.org/wiki/Stock_keeping_unit). This string must not contain newline characters.
    </Description>
  </Field>

  <Field name="identifiers" required={false}>
    <Description>
      Your own identifiers for this sales order item.
    </Description>
    <Type>
      [identifiers object](./identifiers-object.mdx)
    </Type>
  </Field>

  <Field name="description" type="string" required={false}>
    <Description>
      The carrier's description of the charge, not specific to the user. This string must not contain newline characters.
    </Description>
  </Field>



  <Field name="itemURL" required={false}>
    <Type>
      [URL](https://nodejs.org/api/url.html) or string
    </Type>
    <Description>
      The URL of a webpage where the customer can view the order item. Must be a valid HTTP or HTTPS url.
    </Description>
  </Field>

  <Field name="thumbnailURL" required={false}>
    <Type>
      [URL](https://nodejs.org/api/url.html) or string
    </Type>
    <Description>
      The URL of a webpage where the customer can view an image of the order item.
    </Description>
  </Field>

  <Field name="location" required={false}>
    <Type>
      string
    </Type>
    <Description>
      The location where the product can be found in a warehouse.
    </Description>
  </Field>

</Reference>
