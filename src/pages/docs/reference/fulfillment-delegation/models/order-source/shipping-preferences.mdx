ShippingPreferences
===============================================
This represents the shipping preferences specified for an order

<Reference>
  <Field name="digital_fulfillment" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not the item being delivered is a digital good or not
    </Description>
  </Field>
  <Field name="additional_handling" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this shipment will require additional handling
    </Description>
  </Field>
  <Field name="bill_duties_to_sender" type="boolean" nullable={true}>
    <Description>
      Indicates whether duties should be billed to the sender of the package
    </Description>
  </Field>
  <Field name="do_not_prepay_postage" type="boolean" nullable={true}>
    <Description>
      Indicates whether you must pay for postage at drop off, for carriers that usually require prepaid postage
    </Description>
  </Field>
  <Field name="gift" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this order is a gift
    </Description>
  </Field>
  <Field name="has_alcohol" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this order contains alcohol
    </Description>
  </Field>
  <Field name="insurance_requested" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not insurance has been requested for shipping this order
    </Description>
  </Field>
  <Field name="non_machinable" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this order is nonmachinable (must be sorted outside of the standard, automated mail process)
    </Description>
  </Field>
  <Field name="saturday_delivery" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this order should be delivered on a Saturday
    </Description>
  </Field>
  <Field name="show_postage" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not to allow display of postage paid on the shipping label
    </Description>
  </Field>
  <Field name="suppress_email_notify" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not to supress email notifications to the buyer
    </Description>
  </Field>
  <Field name="suppress_marketplace_notify" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not to supress email notifications to the seller
    </Description>
  </Field>
  <Field name="deliver_by_date" type="string" nullable={true}>
    <Description>
      The (ISO 8601) datetime (UTC) associated with when the order needs to be delivered by - @example "2021-03-31T18:21:14.858Z"
    </Description>
  </Field>
  <Field name="hold_until_date" type="string" nullable={true}>
    <Description>
      The (ISO 8601) datetime (UTC) associated with how long to hold the order - @example "2021-03-31T18:21:14.858Z"
    </Description>
  </Field>
  <Field name="ready_to_ship_date" type="string" nullable={true}>
    <Description>
      The (ISO 8601) datetime (UTC) associated with when the order is ready to ship - @example "2021-03-31T18:21:14.858Z"
    </Description>
  </Field>
  <Field name="ship_by_date" type="string" nullable={true}>
    <Description>
      The (ISO 8601) datetime (UTC) associated with when the order needs to be shipped by - @example "2021-03-31T18:21:14.858Z"
    </Description>
  </Field>
  <Field name="preplanned_fulfillment_id" type="string" nullable={true}>
    <Description>
      The identifer assigned by a fulfillment planning system at checkout (Delivery Options)
    </Description>
  </Field>
  <Field name="shipping_service" type="string" nullable={true}>
    <Description>
      The requested shipping service for this fulfillment
    </Description>
  </Field>
  <Field name="package_type" type="string" nullable={true}>
    <Description>
      The requested package type for this fulfillment
    </Description>
  </Field>
  <Field name="insured_value" type="number" nullable={true}>
    <Description>
      The amount of money being request for insurance on this shipment
    </Description>
  </Field>
  <Field name="is_premium_program" type="boolean" nullable={true}>
    <Description>
      true if the order was placed under the terms of the order source's premium program (Amazon Prime, Walmart+, etc)
    </Description>
  </Field>
  <Field name="premium_program_name" type="string" nullable={true}>
    <Description>
      The name of the premium program, if any (This is for informational purposes - Consumers should base all logic on is_premium_program flag)
    </Description>
  </Field>
  <Field name="requested_warehouse" type="string" nullable={true}>
    <Description>
      The warehouse name associated with the requested warehouse
    </Description>
  </Field>
  <Field name="documents" nullable={true}>
    <Type>
      [Document](./../documents/document.mdx)[]
    </Type>
    <Description>
      Any documents relevant to shipping that the order source provides
    </Description>
  </Field>
</Reference>
