Address
===============================================
Basic structure for an address

<Reference>
  <Field name="name" type="string" nullable={true}>
    <Description>
      The full name of the contact associated with this address
    </Description>
  </Field>
  <Field name="company" type="string" nullable={true}>
    <Description>
      The name of the company associated with this address
    </Description>
  </Field>
  <Field name="phone" type="string" nullable={true}>
    <Description>
      The phone number associated with this address
    </Description>
  </Field>
  <Field name="address_line_1" type="string" nullable={true}>
    <Description>
      The first line of the address
    </Description>
  </Field>
  <Field name="address_line_2" type="string" nullable={true}>
    <Description>
      The second line of the address
    </Description>
  </Field>
  <Field name="address_line_3" type="string" nullable={true}>
    <Description>
      The third line of the address
    </Description>
  </Field>
  <Field name="city" type="string" nullable={true}>
    <Description>
      The city associated with this address
    </Description>
  </Field>
  <Field name="state_province" type="string" nullable={true}>
    <Description>
      The state, province, or municipality of the address
    </Description>
  </Field>
  <Field name="postal_code" type="string" nullable={true}>
    <Description>
      The postal code associated with this address
    </Description>
  </Field>
  <Field name="country_code" type="string" nullable={true}>
    <Description>
      The two character ISO 3166 country code of this address @example ("US","MX","CA")
    </Description>
  </Field>
  <Field name="residential_indicator" nullable={true}>
    <Type>
      string | [ResidentialIndicator](./residential-indicator.mdx)
    </Type>
    <Description>
      Indicates this is a residential or commercial address.
    </Description>
  </Field>
  <Field name="is_verified" type="boolean" nullable={true}>
    <Description>
      Indicates whether or not this address has been verified using an Address Verification Service. Values: true, false, and null
    </Description>
  </Field>
  <Field name="pickup_location" nullable={true}>
    <Type>
      [PickupLocation](./pickup-location.mdx)
    </Type>
    <Description>
      Information details related to a pickup location if applicable
    </Description>
  </Field></Reference>
