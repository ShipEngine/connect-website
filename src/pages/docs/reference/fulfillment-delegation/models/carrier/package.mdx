Package
===============================================
This represents a package

  </Field>
  <Field name="dimension_details" nullable={true}>

    </Description>
  </Field>
  <Field name="weight_details" nullable={true}>
    <Type>
      [WeightDetails](./../units/weight-details.mdx)

    </Description>
  </Field>
  <Field name="insured_value" nullable={false}>
    <Type>

  </Field>
  <Field name="label_messages" nullable={true}>

      [LabelMessage](./labels/label-message.mdx)

    <Description>

  <Field name="customs" nullable={true}>
    <Type>
