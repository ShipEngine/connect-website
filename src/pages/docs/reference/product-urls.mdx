ProductUrls
===============================================
The urls associated with a product

<Reference>
  <Field name="image_url" type="string" nullable={true}>
    <Description>
      A link to the image for a product if available
    </Description>
  </Field>
  <Field name="product_url" type="string" nullable={true}>
    <Description>
      A link to the product page if available
    </Description>
  </Field>
  <Field name="thumbnail_url" type="string" nullable={true}>
    <Description>
      A link to the image for use in platform thumbnails
    </Description>
  </Field>
</Reference>
