RejectSalesOrderItemsResult
===============================================
Reject result item

<Reference>
  <Field name="confimation_code" type="string" nullable={true}>
    <Description>
      An identifier provided by the order source to reject the items on the fulfillment
    </Description>
  </Field>
  <Field name="failure_reason" type="string" nullable={true}>
    <Description>
      Details given by the order source if the reject action fails
    </Description>
  </Field>
  <Field name="notification_id" type="string" nullable={false}>
    <Description>
      Identifier for the requested reject notification
    </Description>
  </Field>
  <Field name="status" nullable={false}>
    <Type>
      string
    </Type>
    <Description>
      Result status from the reject action performed on the item. Only supported values are 'success' or 'failure'.
    </Description>
  </Field>
</Reference>
